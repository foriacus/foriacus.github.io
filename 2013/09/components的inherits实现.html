components的原型继承实现

2013-09-19

components的inherits

components是活跃开发者及express框架作者TJ主导的一个前端模块化开发框架，前端颗粒可以小到indexof方法，大到样式复杂的UI组件。在阅读inhertis模块代码时，又重新理解了原形继承的概念。

源码是这样的：

module.exports = function(a, b){
  var fn = function(){};
  fn.prototype = b.prototype;
  a.prototype = new fn;
  a.prototype.constructor = a;
};

Example:

var inherit = require('inherit');
function Animal() {}
function Pet() {}
inherit(Pet, Animal);

简单的分析的源码我们知道，在执行a.prototype = new fn;后，a的实例的constructor已成为b，而不是a，要解决这个问题，就要执行a.prototype.constructor = a;,重新绑定a的构造函数。但是即使是这样，a仍然失去了自己原本的原型，而全部被b的原型取代。所以components的inherts方法是有局限性的。Pet伪类继承Animal是可以，但是前提是Pet也只能作为一个空对象，它自己的原型方法会被清除。

<!Doctype html>
<html>
    <head>
        <meta charset="utf-8">
        <title>测试页面</title>
    </head>
    <body>
        <script>      
            function inherits(a, b) { 
                var Fn = function() {};
                Fn.prototype = b.prototype;
                //此处重新定义函数a的prototype
                a.prototype = new Fn;
                a.prototype.constructor = a;
            }

            function Person(name) {
                this.name = name;
            }

            Person.prototype.getName = function() { 
                return this.name; 
            }

            var p = new Person('Jack')

            //alert(p.constructor === Person);
            //alert(Person.prototype.constructor === Person);

            function Teacher(name) {
                this.name = name
            }

            Teacher.prototype.teach = function () {
                return 'please speak follow me';
            }

            inherits(Teacher, Person);
            var xiaoming = new Teacher('xiaoming');
            alert(xiaoming.constructor === Teacher);  //true
            alert(xiaoming.hasOwnProperty('teach'));  //false 
        </script>
    </body>
</html>
